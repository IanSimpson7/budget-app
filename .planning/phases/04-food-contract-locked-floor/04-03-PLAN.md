FloorGap =
  | { type: 'unpriced-ingredient'; ingredientName: string }
  | { type: 'undefined-meal'; mealName: string }
  | { type: 'unset-flat-cost'; mealName: string }
  | { type: 'stale-plan'; lastKnownDate: string | null }

CostEngineInput:
  scheduledMeals: string[]        // normalized meal names covering today's month (from ParsedPlan.meals)
  mealDefinitions: MealDefinition[]
  unitCostMap: UnitCostEntry[]
  portionModel: PortionEntry[]
  daysInMonth: number             // days in the CURRENT calendar month (~28-31).
                                  // CONTRACT (I-01): the CALLER (foodFloorAtom, Plan 04) MUST compute this
                                  // from the current date at READ time, never a memoized write-time value,
                                  // so the monthly floor is correct for whatever month the app is opened in.
                                  // computeFloor is pure: it trusts the daysInMonth it is given.
  windowDays: number              // distinct days the current plan(s) cover (for the daily-average)

CostEngineResult:
  floor: number                   // D-06: dailyAverage times daysInMonth
  gaps: FloorGap[]
  isClean: boolean                // gaps.length === 0

computeFloor(input: CostEngineInput): CostEngineResult
isPlanCurrent(plans: ParsedPlan[], today: string): boolean   // some window covers today (D-08)
fallbackFloor(meta: FoodFloorMeta): number                   // max(lastComputed, allTimeHighWater) (D-07)

FALLBACK_CEILING_PER_MEAL: number   // OPEN-QUESTION-1 RESOLUTION = 15.00 (module constant)
  // Rationale: Qdoba bowl ~ $11; a full prep meal ~ $3-5; $15 is conservative-high per meal.
  // A static constant (NOT "most-expensive-defined-meal") so the fallback is deterministic
  // even when no meals are defined yet (early setup). Per RESEARCH Open Question 1 / A4.
